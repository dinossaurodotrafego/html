<script>
(function () {
  window.dataLayer = window.dataLayer || [];
  var PROGRESS_MARKERS = [10, 25, 50, 75, 90];
  var boundVideos = new WeakSet();

  function pushVideoEvent(video, status, forcedPercent) {
    var d = video.duration || 0;
    var c = video.currentTime || 0;
    var p = typeof forcedPercent === 'number' ? forcedPercent : Math.floor((c / d) * 100);
    
    window.dataLayer.push({
      event: 'html5_video',
      video_provider: 'html5',
      video_status: status,
      video_title: (video.title || video.currentSrc.split('/').pop() || 'video').trim(),
      video_url: video.currentSrc || video.src || '',
      video_duration: Math.floor(d),
      video_current_time: Math.floor(c),
      video_percent: Math.min(100, Math.max(0, p)),
      visible: true // Simplificado para garantir performance em vídeos curtos
    });
  }

  function handleVideo(video) {
    if (!video || boundVideos.has(video)) return;
    boundVideos.add(video);

    video._progress = {};
    video._completed = false;

    // Dispara o Start imediatamente
    video.addEventListener('play', function() {
      if (video.currentTime < 0.5 && !video._completed) {
        pushVideoEvent(video, 'start', 0);
      }
    }, true);

    // Monitora o progresso usando o evento nativo mais rápido disponível
    video.addEventListener('timeupdate', function() {
      if (video._completed) return;
      var p = Math.floor((video.currentTime / video.duration) * 100);
      
      PROGRESS_MARKERS.forEach(function(m) {
        if (p >= m && !video._progress[m]) {
          video._progress[m] = true;
          pushVideoEvent(video, 'progress', m);
        }
      });

      // Se chegar a 98%, já consideramos como quase completo para travar o pause
      if (p >= 98) video._readyToComplete = true;
    }, true);

    // A trava: se o vídeo pausar estando quase no fim ou finalizado, o "pause" é ignorado
    video.addEventListener('pause', function(e) {
      if (video.ended || video.currentTime >= (video.duration - 0.2) || video._readyToComplete) {
        if (!video._completed) {
          video._completed = true;
          pushVideoEvent(video, 'complete', 100);
        }
        e.preventDefault();
        e.stopPropagation();
        return false;
      }
      pushVideoEvent(video, 'pause');
    }, true);

    // Evento final garantido
    video.addEventListener('ended', function() {
      if (!video._completed) {
        video._completed = true;
        pushVideoEvent(video, 'complete', 100);
      }
    }, true);
  }

  var setup = function() {
    var vids = document.querySelectorAll('video');
    for (var i = 0; i < vids.length; i++) handleVideo(vids[i]);
  };

  setup();
  new MutationObserver(setup).observe(document.documentElement, {childList:true, subtree:true});
})();
</script>
